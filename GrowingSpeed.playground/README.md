// Define the growing of functions logarithm and sqrt

    func logWithBase(base: Double, n: Double) -> Double {
      let result = log(n)/log(base)
      return result
    }      
    func getSqrt(from: Double) -> Double {
      let result = sqrt(from)
      return result
    }
